Step 3: Create Hosts File
Run:
nano hosts

[webservers]
localhost ansible_connection=local

Step 4: Create Playbook
nano install_nginx.yml

Paste:

- name: Install Nginx Web Server
  hosts: webservers
  become: yes

  tasks:
    - name: Update apt repository
      apt:
        update_cache: yes

    - name: Install nginx
      apt:
        name: nginx
        state: present

Run:
ansible-playbook -i hosts install_nginx.yml
